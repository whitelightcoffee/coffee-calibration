三種記錄（淺焙SOE / 中焙SOE / 手沖）共用同一個 URL，透過 type 欄位寫入不同 Sheet
Apps Script 程式碼（貼入後部署）
function doPost(e) { var p = e.parameter; var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheetName = p.type === 'light' ? '淺焙SOE' : p.type === 'medium' ? '中焙SOE' : '手沖'; var sheet = ss.getSheetByName(sheetName) || ss.insertSheet(sheetName); if (sheet.getLastRow() === 0) { if (p.type === 'pour') { sheet.appendRow(['時間','夥伴','時段','氣溫℃','濕度%','水溫℃','研磨刻度','Tonik下','LyLac下','豆名','批次','Agtron豆','Agtron粉','密度','CVA香氣','CVA風味','CVA餘韻','CVA酸度','CVA甜感','CVA觸感','CVA整體印象','CVA豆子類型','CVA香氣詞','CVA風味詞','CVA餘韻詞','CVA酸度詞','CVA觸感詞','CVA香氣補充','CVA風味補充','CVA餘韻補充','CVA酸度補充','CVA觸感補充','備注']); } else { sheet.appendRow(['時間','夥伴','時段','氣溫℃','濕度%','水溫℃','粉量g','萃取秒','咖啡液重g','Espresso比例','豆名','批次','Agtron豆','Agtron粉','密度','酸值%','甜感%','深調%','狀態','標籤','備注']); } } if (p.type === 'pour') { sheet.appendRow([p.ts,p.staff,p.time,p.temp,p.humid,p.wtemp,p.grind,p.tonik,p.lylac,p.beanName,p.batchNo,p.agtronBean,p.agtronGrind,p.density,p.cvaAroma,p.cvaFlavor,p.cvaAftertaste,p.cvaAcidity,p.cvaSweetness,p.cvaMouthfeel,p.cvaOverall,p.cvaProfile,p.cvaAromaTags,p.cvaFlavorTags,p.cvaAftertasteTags,p.cvaAcidityTags,p.cvaMouthfeelTags,p.cvaAromaText,p.cvaFlavorText,p.cvaAftertasteText,p.cvaAcidityText,p.cvaMouthfeelText,p.note]); } else { sheet.appendRow([p.ts,p.staff,p.time,p.temp,p.humid,p.wtemp,p.dose,p.brewTime,p.yield,p.ratio,p.beanName,p.batchNo,p.agtronBean,p.agtronGrind,p.density,p.acid,p.sweet,p.dark,p.status,p.tags,p.note]); } return ContentService.createTextOutput('OK'); } function doGet(e) { return ContentService.createTextOutput('OK'); }
