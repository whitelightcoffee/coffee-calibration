function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName('記錄') || ss.insertSheet('記錄');
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(['時間','夥伴','時段','氣溫℃','濕度%','豆名','批次','Agtron豆','Agtron粉','密度','酸值%','甜感%','深調%','狀態','標籤','備注']);
  }
  var d = JSON.parse(e.postData.contents);
  sheet.appendRow([d.ts,d.staff,d.time,d.temp,d.humid,d.beanName,d.batchNo,d.agtronBean,d.agtronGrind,d.density,d.acid,d.sweet,d.dark,d.status,d.tags,d.note]);
  return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON);
}
